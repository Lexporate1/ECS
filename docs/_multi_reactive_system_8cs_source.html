62  /// Will call Execute(entities) with changed entities

65  for (int i = 0; i < _collectors.Length; i++) {
66  var collector = _collectors[i];
67  if (collector.count != 0) {
68  foreach (var e in collector.GetCollectedEntities<TEntity>()) {

85  }
